#### 一百钱买一百只鸡,公鸡5文一只,母鸡3文一只,小鸡1文三只;问有公鸡母鸡小鸡各多少只?
~~~
for($a=1;$a<=18;$a++){//18是因为全买公鸡
	for($b=1;$b<=31;$b++){
		for($c=1;$c<=100;$c++){
			if(($a+$b+$c==100)&&(5*$a+3*$b+$c/3==100)){
				echo "公鸡{$a}只,母鸡{$b}只,小鸡{$c}只<br>";
			}
		}
	}
}

递归
~~~
function jie($n){
	if($n==1){
		return 1;
	}else{
		return jie($n-1)*$n;
	}
}

echo jie(10);
